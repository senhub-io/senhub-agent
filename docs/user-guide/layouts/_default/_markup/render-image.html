{{- /* Project override of the theme image render hook. Hugo does not prepend the baseURL subpath to root-absolute markdown destinations (e.g. /images/x) when canonifyURLs is off, so those images 404 under a subpath deploy (https://agent.senhub.io/docs/). Derive the base path from site.BaseURL and prepend it to root-absolute paths; external and relative URLs are left as-is. Keeps the theme's BookPortableLinks behaviour. */ -}} {{- $d := .Destination -}} {{- if .Page.Site.Params.BookPortableLinks -}} {{- $d = partial "docs/links/portable-image" . -}} {{- else if hasPrefix $d "/" -}} {{- $base := strings.TrimSuffix "/" (urls.Parse site.BaseURL).Path -}} {{- $d = printf "%s%s" $base $d -}} {{- end -}} {{- /**/ -}}
